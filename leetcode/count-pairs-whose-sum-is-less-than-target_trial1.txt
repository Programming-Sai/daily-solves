class Solution:
    def countPairs(self, nums: List[int], target: int) -> int:
        nums.sort()
        l, r, num_pairs = 0, len(nums)-1, 0
        while l < r:
            if nums[l] + nums[r] < target:
                num_pairs += (r-l)
                l += 1
            else:
                r -= 1
        return num_pairs
