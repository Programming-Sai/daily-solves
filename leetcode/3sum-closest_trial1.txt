class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        n = len(nums)
        min_diff, closest_sum = float("inf"), 0
        for k in range(n):
            l = k + 1
            r = n - 1
            while l < r:
                sum_ = nums[l] + nums[k] + nums[r]
                diff = abs(sum_ - target)
                if diff < min_diff: min_diff = diff; closest_sum = sum_
                if sum_ == target:
                    return sum_
                elif sum_ < target:
                    l += 1
                else:
                    r -= 1
        return closest_sum
