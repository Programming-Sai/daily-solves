class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        l, r, n, max_len, max_freq, freq = 0, 0, len(s), 0, 0, defaultdict(int)
        while r < n:
            freq[s[r]] += 1
            max_freq = max(max_freq, freq[s[r]])
            while l < n and ((r-l+1) - max_freq) > k:
                freq[s[l]] -= 1
                l += 1
            max_len = max(max_len, r-l+1)
            r += 1
        return max_len
