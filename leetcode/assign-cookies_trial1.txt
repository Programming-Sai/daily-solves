class Solution:
    def findContentChildren(self, g: List[int], s: List[int]) -> int:
        g.sort(), s.sort()
        i, j, l_i, l_j, count = 0, 0, len(g), len(s), 0
        while i < l_i and j < l_j:
            if s[j] >= g[i]:
                count += 1
                i += 1
            j += 1
        return count
