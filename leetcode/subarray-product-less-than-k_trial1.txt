class Solution:
    def numSubarrayProductLessThanK(self, nums: List[int], k: int) -> int:
        if k <= 1:
            return 0
        l, r, n, count, product = 0, 0, len(nums), 0, 1
        while r < n:
            product *= nums[r]
            while product >= k:
                product //= nums[l]
                l += 1            
            count += (r - l + 1)
            r += 1
        return count
