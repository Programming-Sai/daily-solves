class Solution:
    def carFleet(self, target: int, position: List[int], speed: List[int]) -> int:
        cars = sorted([(position[i], speed[i]) for i in range(len(speed))], key=lambda x:x[0], reverse=True)
        stack = []
        for car in cars:
            time = float(target - car[0]) / car[1]
            if stack and time <= stack[-1]: continue
            stack.append(time)
        return len(stack)
