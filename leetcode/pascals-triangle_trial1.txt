class Solution:
    def generate(self, numRows: int) -> List[List[int]]:
        if  numRows == 1: return [[1]]
        triangle = self.generate(numRows - 1)
        last_prev_row = triangle[-1]
        curr = [1]
        for i in range(1, len(last_prev_row)):
            curr.append(last_prev_row[i-1] + last_prev_row[i])
        curr.append(1)
        triangle.append(curr)
        return triangle
