class MyCircularDeque:

    def __init__(self, k: int):
        self.deque = [None for _ in range(k)]
        self.front = 0
        self.end = 0
        self.capacity = k
        self.size = 0

    def insertFront(self, value: int) -> bool:
        if self.isFull(): return False
        self.front = (self.front - 1 + self.capacity) % self.capacity
        self.deque[self.front] = value
        self.size += 1
        return True

    def insertLast(self, value: int) -> bool:
        if self.isFull(): return False
        self.deque[self.end] = value
        self.end = (self.end + 1) % self.capacity
        self.size += 1
        return True

    def deleteFront(self) -> bool:
        if self.isEmpty(): return False
        self.front = (self.front + 1) % self.capacity
        self.size -= 1
        if self.isEmpty(): self.front = self.end = 0
        return True

    def deleteLast(self) -> bool:
        if self.isEmpty(): return False
        self.end = (self.end - 1 + self.capacity) % self.capacity
        self.size -= 1
        if self.isEmpty(): self.front = self.end = 0
        return True        

    def getFront(self) -> int:
        if self.isEmpty(): return -1
        return self.deque[self.front]

    def getRear(self) -> int:
        if self.isEmpty(): return -1
        last_idx = (self.end - 1 + self.capacity) % self.capacity
        return self.deque[last_idx]

    def isEmpty(self) -> bool:
        return self.size == 0

    def isFull(self) -> bool:
        return self.size == self.capacity
