class NumArray:

    def __init__(self, nums: List[int]):
        self.prefix_nums = [0] + list(accumulate(nums))

    def sumRange(self, left: int, right: int) -> int:
        return self.prefix_nums[right + 1] - self.prefix_nums[left]
