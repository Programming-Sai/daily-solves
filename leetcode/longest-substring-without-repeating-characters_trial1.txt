class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        l, r, window, max_len, n = 0, 0, set(), 0, len(s)

        while r < n:
            while s[r] in window:
                window.remove(s[l])
                l += 1
            window.add(s[r])
            max_len = max(max_len, (r - l + 1))
            r += 1
        return max_len
