class Solution:
    def getRow(self, rowIndex: int) -> List[int]:
        curr = [1]
        if rowIndex == 0: return curr
        prev_row = self.getRow(rowIndex-1)
        for i in range(1, len(prev_row)):
            curr.append(prev_row[i-1] + prev_row[i])
        curr.append(1)
        return curr
