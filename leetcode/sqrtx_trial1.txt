class Solution:
    def mySqrt(self, x: int) -> int:
        if x < 2: return x
        l, r, res = 1, x//2, -1
        while l <= r:
            mid = l + (r-l)//2
            if mid ** 2 <= x:
                res = mid
                l = mid + 1
            else:
                r = mid - 1
        return res
