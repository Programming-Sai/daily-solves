class Solution:
    def checkSubarraySum(self, nums: List[int], k: int) -> bool:
        seen, total = {0: -1}, 0
        for i, num in enumerate(nums):
            total += num
            r = total % k
            if r in seen:
                if i - seen[r] > 1:
                    return True
            else:
                seen[r] = i
        return False
