class Solution:
    def totalFruit(self, fruits: List[int]) -> int:
        l, r, n, window_set, max_len, k = 0, 0, len(fruits), Counter(), float("-inf"), 2
        while r < n:
            window_set[fruits[r]] += 1
            while len(window_set) > k:
                window_set[fruits[l]] -= 1
                if window_set[fruits[l]] < 1: del window_set[fruits[l]]
                l += 1
            max_len = max(max_len, r-l+1)
            r += 1
        return max_len
