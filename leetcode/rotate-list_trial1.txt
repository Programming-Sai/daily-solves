# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if not head or k == 0:
            return head
        length, tail, curr = 0, None, head
        while curr:
            length += 1
            tail = curr
            curr = curr.next
        k = k % length
        if k == 0:return head
        tail.next = head
        new_tail_pos, curr, new_tail, new_head = length - k - 1, head, None, None
        while curr:
            if new_tail_pos == 0:
                new_tail = curr
                new_head = curr.next
                new_tail.next = None
                return new_head
            curr = curr.next
            new_tail_pos -= 1
