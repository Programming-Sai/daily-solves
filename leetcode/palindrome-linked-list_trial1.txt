# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def rev(self, head):
        curr, prev = head, None
        while curr:
            new = curr.next
            curr.next = prev
            prev = curr
            curr = new
        return prev

    def isPalindrome(self, head: Optional[ListNode]) -> bool:
        fast, slow, curr, count = head, head, head, 0
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
            count += 1
        if fast: slow = slow.next
        r = self.rev(slow)
        while r:
            if curr.val != r.val:
                return False
            curr = curr.next
            r = r.next
        return True
