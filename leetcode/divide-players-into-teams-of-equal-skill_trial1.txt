class Solution:
    def dividePlayers(self, skill: List[int]) -> int:
        skill.sort()
        l, r, n, sum_of_product, init_sum, chem_score = 0, len(skill)-1, len(skill), 0, skill[0]+skill[-1], 1
        while l < r:
            if skill[l] + skill[r]  != init_sum:
                return -1
            sum_of_product += skill[l] * skill[r]
            l += 1
            r -= 1
        return sum_of_product
